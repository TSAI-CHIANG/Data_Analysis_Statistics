We are 95% confident the true value of the mean house price is between 269.006 and 307.6099 ($1000).

raw<- read.csv("hprice.csv")
summary(raw$price)
##    Min. 1st Qu.  Median    Mean 3rd Qu.    Max. 
##   111.0   230.0   264.0   288.3   316.0   725.0
E1a <- function(x){
  # sample mean
  mu <- mean(x)
  
  # sample standard deviation (standard error)
  se <- sd(x)/sqrt(length(x))
  
  # test
  test <- t.test(x, 
                 alternative = c("two.sided"),
                 mu = 0, # H0
                 conf.level = 0.95) # alpha = 0.05
  
  # 95% confidence interval
  lower <- round(test$conf.int[1], digit = 4)
  upper <- round(test$conf.int[2], digit = 4)
  CI <- paste(lower, "-"  ,upper)
  
  Table <- data.frame(mu, se, CI)
  colnames(Table) <- c("Mean", "Standard Error", "95% Confidence Interval")
  
  Table
  
}
E1a(raw$price)
